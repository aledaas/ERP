openssl req \
    -newkey rsa:2048 \
    -x509 \
    -nodes \
    -keyout systemerp.local.key \
    -new \
    -out systemerp.local.crt \
    -subj /CN=\*.systemerp.local \
    -reqexts SAN \
    -extensions SAN \
    -config <(cat /System/Library/OpenSSL/openssl.cnf \
        <(printf '[SAN]\nsubjectAltName=DNS:\*.systemerp.local')) \
    -sha256 \
    -days 3650
